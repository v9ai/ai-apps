### Practical Considerations for Error Handling

Trafilatura handles most malformed HTML gracefully but edge cases remain:

```python
import logging
import chardet
import trafilatura

logger = logging.getLogger(__name__)


def safe_extract(raw_bytes: bytes, url: str) -> str | None:
    """Extract clean text with encoding detection and error recovery."""
    # 1. Detect encoding -- chardet as fallback
    declared = trafilatura.utils.detect_encoding(raw_bytes)
    if declared is None:
        result = chardet.detect(raw_bytes)
        declared = result.get("encoding", "utf-8")
        confidence = result.get("confidence", 0.0)
        if confidence < 0.5:
            logger.warning(
                "Low encoding confidence %.2f for %s, falling back to utf-8",
                confidence, url,
            )
            declared = "utf-8"

    try:
        html = raw_bytes.decode(declared, errors="replace")
    except (UnicodeDecodeError, LookupError) as exc:
        logger.warning("Decode failed for %s (%s): %s", url, declared, exc)
        html = raw_bytes.decode("utf-8", errors="replace")

    # 2. Extract with trafilatura
    try:
        text = trafilatura.extract(
            html,
            include_tables=True,
            include_links=True,
            favor_recall=True,
        )
    except Exception as exc:
        logger.error("trafilatura extraction failed for %s: %s", url, exc)
        return None

    # 3. Validate output
    if not text or len(text.strip()) < 50:
        logger.info("Extraction too short for %s (%d chars) -- skipping",
                     url, len(text.strip()) if text else 0)
        return None

    return text
```

Key failure modes:
- **Charset mismatch**: declared charset != actual encoding. `chardet` catches
  most cases; the `errors="replace"` fallback prevents crashes at the cost of
  occasional mojibake.
- **JavaScript-rendered pages**: trafilatura cannot execute JS. These pages
  produce empty or nav-only extractions. Detect via `len(text) < 50` guard.
- **Extremely large HTML** (>5 MB): can cause memory spikes. Truncate
  `raw_bytes` before extraction.

### Confidence Calibration

The NER model outputs softmax probabilities per token. We use per-type
thresholds to gate entity acceptance:

| Entity Type | Threshold | Rationale                               |
|-------------|-----------|------------------------------------------|
| ORG         | 0.75      | High data volume, well-calibrated        |
| PERSON      | 0.75      | High data volume, well-calibrated        |
| LOCATION    | 0.60      | Less training data, lower model certainty|
| PRODUCT     | 0.60      | Less training data, noun overlap         |

**Calibration caveat:** These thresholds were selected by maximising F1 on the
validation split. They have **not** been verified with post-hoc calibration
methods (temperature scaling, Platt scaling, isotonic regression). On
out-of-distribution text (e.g., non-English, scientific abstracts, social media)
the relationship between softmax probability and true correctness may diverge
significantly.

**Recommendation:** Run temperature scaling on a held-out calibration set of
~500 documents before deploying to a new domain. The procedure:

```python
import logging
import torch
import torch.nn as nn
from torch.utils.data import DataLoader

logger = logging.getLogger(__name__)


class TemperatureScaler(nn.Module):
    """Learn a single temperature parameter on held-out logits."""

    def __init__(self):
        super().__init__()
        self.temperature = nn.Parameter(torch.ones(1) * 1.5)

    def forward(self, logits):
        return logits / self.temperature


def calibrate_temperature(model, calibration_loader: DataLoader, lr=0.01, max_iter=50):
    """Fit temperature on calibration set, return scaler."""
    scaler = TemperatureScaler()
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.LBFGS([scaler.temperature], lr=lr, max_iter=max_iter)

    all_logits, all_labels = [], []
    with torch.no_grad():
        for batch in calibration_loader:
            try:
                outputs = model(**batch["inputs"])
                all_logits.append(outputs.logits)
                all_labels.append(batch["labels"])
            except Exception as exc:
                logger.error("Calibration forward pass failed: %s", exc)

    logits = torch.cat(all_logits)
    labels = torch.cat(all_labels)

    def closure():
        optimizer.zero_grad()
        loss = criterion(scaler(logits), labels)
        loss.backward()
        return loss

    optimizer.step(closure)
    logger.info("Calibrated temperature: %.4f", scaler.temperature.item())
    return scaler
```

After calibration, replace raw softmax with `softmax(logits / T)` where `T` is
the learned temperature. Typical values: T in [1.2, 2.5] for fine-tuned BERT.

### NER Error Handling

```python
import logging
import signal

logger = logging.getLogger(__name__)

NER_TIMEOUT_SECONDS = 30


class NERTimeoutError(Exception):
    pass


def _timeout_handler(signum, frame):
    raise NERTimeoutError("NER inference exceeded timeout")


def run_ner_with_timeout(model, tokenizer, text: str, url: str):
    """Run NER inference with a per-document timeout.

    Returns list of entity dicts or empty list on failure.
    """
    signal.signal(signal.SIGALRM, _timeout_handler)
    signal.alarm(NER_TIMEOUT_SECONDS)
    try:
        inputs = tokenizer(
            text,
            return_tensors="pt",
            truncation=True,
            max_length=512,
        )
        outputs = model(**inputs)
        logits = outputs.logits
        probabilities = logits.softmax(dim=-1)
        predictions = logits.argmax(dim=-1)

        entities = []
        for idx, (pred, prob) in enumerate(
            zip(predictions[0], probabilities[0])
        ):
            label = model.config.id2label[pred.item()]
            confidence = prob[pred].item()
            if label == "O":
                continue

            # Per-type threshold
            entity_type = label.split("-")[-1] if "-" in label else label
            threshold = _get_threshold(entity_type)
            if confidence < threshold:
                continue

            entities.append({
                "token_idx": idx,
                "label": label,
                "confidence": confidence,
            })

        return entities

    except NERTimeoutError:
        logger.error("NER timeout (%ds) for %s", NER_TIMEOUT_SECONDS, url)
        return []
    except Exception as exc:
        logger.error("NER inference failed for %s: %s", url, exc)
        return []
    finally:
        signal.alarm(0)


def _get_threshold(entity_type: str) -> float:
    """Return confidence threshold for an entity type."""
    thresholds = {
        "ORG": 0.75,
        "PERSON": 0.75,
        "LOCATION": 0.60,
        "LOC": 0.60,
        "PRODUCT": 0.60,
    }
    return thresholds.get(entity_type.upper(), 0.75)
```

```python
import logging
import spacy
from transformers import BertForSequenceClassification, BertTokenizer

logger = logging.getLogger(__name__)

RELATION_LABELS = [
    "company_in_industry", "company_launched_product",
    "company_acquired_company", "person_joined_company",
    "company_raised_funding", "company_partnered_company",
    "person_founded_company", "company_expanded_location",
    "person_left_company", "company_ipo",
    "company_hired_person", "company_merged_company",
    "none",
]


class RelationExtractor:
    def __init__(self, model_path: str, dep_path_max: int = 4):
        self.dep_path_max = dep_path_max
        try:
            self.nlp = spacy.load("en_core_web_sm")
        except OSError:
            logger.error("spaCy model 'en_core_web_sm' not found -- "
                         "run: python -m spacy download en_core_web_sm")
            raise
        try:
            self.classifier = BertForSequenceClassification.from_pretrained(
                model_path, num_labels=len(RELATION_LABELS)
            )
            self.tokenizer = BertTokenizer.from_pretrained(model_path)
        except Exception as exc:
            logger.error("Failed to load relation model from %s: %s",
                         model_path, exc)
            raise

    def extract(self, text: str, entities: list[dict]) -> list[dict]:
        """Extract relations between entity pairs.

        Returns list of {"subj": str, "pred": str, "obj": str, "confidence": float}.
        """
        if len(entities) < 2:
            return []

        try:
            doc = self.nlp(text)
        except Exception as exc:
            logger.error("spaCy parse failed: %s", exc)
            return []

        relations = []
        for i, ent1 in enumerate(entities):
            for ent2 in entities[i + 1:]:
                try:
                    rel = self._classify_pair(text, ent1, ent2)
                    if rel and rel["pred"] != "none":
                        relations.append(rel)
                except Exception as exc:
                    logger.warning(
                        "Relation classification failed for (%s, %s): %s",
                        ent1.get("name"), ent2.get("name"), exc,
                    )
        return relations

    def _classify_pair(self, text, ent1, ent2):
        """Classify relation between two entities."""
        input_text = f"{text} [SEP] {ent1['name']} [SEP] {ent2['name']}"
        inputs = self.tokenizer(
            input_text,
            return_tensors="pt",
            truncation=True,
            max_length=512,
        )
        outputs = self.classifier(**inputs)
        probs = outputs.logits.softmax(dim=-1)[0]
        pred_idx = probs.argmax().item()
        confidence = probs[pred_idx].item()

        if confidence < 0.5:
            return None

        return {
            "subj": ent1["name"],
            "pred": RELATION_LABELS[pred_idx],
            "obj": ent2["name"],
            "confidence": round(confidence, 4),
        }
```

```python
import json
import logging
import time
import chromadb

logger = logging.getLogger(__name__)

client = chromadb.PersistentClient(path="scrapus_data/chromadb")
page_collection = client.get_or_create_collection(
    name="page_documents",
    metadata={
        "hnsw:space": "cosine",
        "hnsw:construction_ef": 200,
        "hnsw:M": 16,
        "hnsw:search_ef": 200,
    },
)


def store_profile(profile: dict, max_retries: int = 2) -> bool:
    """Store page profile in ChromaDB with retry on transient failure."""
    for attempt in range(max_retries):
        try:
            page_collection.add(
                ids=[profile["url_hash"]],
                embeddings=[profile["content_embedding"]],
                metadatas=[{
                    "url": profile["url"],
                    "title": profile.get("title", ""),
                    "entities_json": json.dumps(profile["entities"]),
                    "relations_json": json.dumps(profile["relations"]),
                    "topics_json": json.dumps(profile["topics"]),
                    "crawl_timestamp": profile["timestamp"],
                    "has_org_entity": any(
                        e["type"] == "ORG" for e in profile["entities"]
                    ),
                    "domain": profile["domain"],
                }],
                documents=[profile["clean_text"][:2000]],
            )
            return True
        except chromadb.errors.DuplicateIDError:
            logger.info("Duplicate ID %s -- updating instead",
                        profile["url_hash"])
            try:
                page_collection.update(
                    ids=[profile["url_hash"]],
                    embeddings=[profile["content_embedding"]],
                    metadatas=[{
                        "url": profile["url"],
                        "title": profile.get("title", ""),
                        "entities_json": json.dumps(profile["entities"]),
                        "relations_json": json.dumps(profile["relations"]),
                        "topics_json": json.dumps(profile["topics"]),
                        "crawl_timestamp": profile["timestamp"],
                        "has_org_entity": any(
                            e["type"] == "ORG" for e in profile["entities"]
                        ),
                        "domain": profile["domain"],
                    }],
                    documents=[profile["clean_text"][:2000]],
                )
                return True
            except Exception as exc:
                logger.error("ChromaDB update failed for %s: %s",
                             profile.get("url", "?"), exc)
                return False
        except Exception as exc:
            logger.error("ChromaDB insert attempt %d failed for %s: %s",
                         attempt + 1, profile.get("url", "?"), exc)
            if attempt < max_retries - 1:
                time.sleep(1.0)
    return False


def check_duplicate(content_embedding, domain: str, threshold: float = 0.05):
    """Check for near-duplicate content. Returns (is_dup, existing_id)."""
    try:
        results = page_collection.query(
            query_embeddings=[content_embedding],
            n_results=5,
            where={"domain": domain},
            include=["metadatas", "distances"],
        )
    except Exception as exc:
        logger.error("Dedup query failed: %s", exc)
        return False, None

    dists = results.get("distances", [[]])[0]
    if dists and dists[0] < threshold:
        dup_id = results["ids"][0][0]
        logger.info("Near-duplicate: %s (dist=%.4f)", dup_id, dists[0])
        return True, dup_id
    return False, None
```

```python
import torch.nn as nn
from transformers import AutoModel, AutoConfig

class SpERT_Scrapus(nn.Module):
    """Span-based joint NER+RE for B2B lead generation"""
    
    def __init__(self, model_name="microsoft/deberta-v3-base"):
        super().__init__()
        self.encoder = AutoModel.from_pretrained(model_name)
        config = AutoConfig.from_pretrained(model_name)
        hidden_size = config.hidden_size
        
        # Span representation layers
        self.span_start = nn.Linear(hidden_size, hidden_size)
        self.span_end = nn.Linear(hidden_size, hidden_size)
        self.span_width = nn.Embedding(50, hidden_size)  # Max span length
        
        # Entity classification head
        self.entity_classifier = nn.Linear(hidden_size * 3, len(ENTITY_TYPES))
        
        # Relation classification head
        self.relation_classifier = nn.Sequential(
            nn.Linear(hidden_size * 6, hidden_size),
            nn.ReLU(),
            nn.Dropout(0.1),
            nn.Linear(hidden_size, len(RELATION_TYPES))
        )
    
    def forward(self, input_ids, attention_mask, spans):
        # Get contextual embeddings
        outputs = self.encoder(input_ids, attention_mask=attention_mask)
        sequence_output = outputs.last_hidden_state
        
        # Extract span representations
        span_reps = []
        for (start, end) in spans:
            start_rep = self.span_start(sequence_output[:, start])
            end_rep = self.span_end(sequence_output[:, end])
            width_rep = self.span_width(end - start)
            
            # Concatenate representations
            span_rep = torch.cat([start_rep, end_rep, width_rep], dim=-1)
            span_reps.append(span_rep)
        
        span_reps = torch.stack(span_reps, dim=1)
        
        # Entity classification
        entity_logits = self.entity_classifier(span_reps)
        
        # Relation classification (pairwise)
        relation_logits = []
        for i in range(len(spans)):
            for j in range(len(spans)):
                if i != j:
                    pair_rep = torch.cat([span_reps[:, i], span_reps[:, j]], dim=-1)
                    rel_logit = self.relation_classifier(pair_rep)
                    relation_logits.append(rel_logit)
        
        return entity_logits, torch.stack(relation_logits, dim=1)
```

```python
                get_url_from_html(html_content),
                clean_text,
                uncertainty["total"],
                compute_diversity_score(clean_text, entities)
            ))
        
        # 5. Return results
        return {
            "entities": entities,
            "relations": relations,
            "uncertainty": uncertainty,
            "needs_annotation": uncertainty["total"] > 0.7
        }
    
    return process_page
```

### Hybrid Architecture

```python
import logging
import spacy
from transformers import BertForSequenceClassification

logger = logging.getLogger(__name__)

RELATION_LABELS = [
    "company_in_industry", "company_launched_product",
    "company_acquired_company", "person_joined_company",
    "company_raised_funding", "company_partnered_company",
    "person_founded_company", "company_expanded_location",
    "person_left_company", "company_ipo", "company_hired_person",
    "company_merged_company", "none",
]


class HybridRelationExtractor:
    def __init__(self, dep_path_max=4):
        self.dep_parser = spacy.load("en_core_web_sm")
        self.bert_classifier = BertForSequenceClassification.from_pretrained(
            "bert-base-cased", num_labels=len(RELATION_LABELS)
        )
        self.dep_path_max = dep_path_max

    def extract_candidate_pairs(self, doc):
        """Dependency-parse filter: keep pairs within dep_path_max hops."""
        candidates = []
        for ent1 in doc.ents:
            for ent2 in doc.ents:
                if ent1 == ent2:
                    continue
                try:
                    path = self._get_dependency_path(ent1, ent2)
                    if len(path) <= self.dep_path_max:
                        candidates.append((ent1, ent2, path))
                except ValueError:
                    # Disconnected spans -- skip
                    continue
        return candidates

    def classify_relation(self, sent, ent1, ent2):
        """BERT classification with entity markers."""
        input_text = (
            f"[CLS] {sent} [SEP] {ent1.text} [SEP] {ent2.text} [SEP]"
        )
        try:
            return self.bert_classifier(input_text)
        except Exception as exc:
            logger.error("Relation classification failed: %s", exc)
            return None

    # ---- internal ----

    @staticmethod
    def _get_dependency_path(ent1, ent2):
        """Return shortest dependency path tokens between two entities."""
        head1 = ent1.root
        head2 = ent2.root
        path = []
        current = head1
        visited = set()
        while current != current.head and current not in visited:
            visited.add(current)
            path.append(current)
            current = current.head
            if current == head2:
                return path
        raise ValueError("No path found")
```

### Batched Inference with Dynamic Padding

```python
import asyncio
import logging
import onnxruntime as ort
from transformers import BertTokenizerFast

logger = logging.getLogger(__name__)


class OptimizedInference:
    def __init__(self, model_path, batch_size=32, max_seq_len=512):
        self.session = ort.InferenceSession(model_path)
        self.batch_size = batch_size
        self.max_seq_len = max_seq_len
        self.tokenizer = BertTokenizerFast.from_pretrained("bert-base-cased")

    def dynamic_batching(self, texts):
        """Sort-then-batch to minimise padding waste."""
        indexed = sorted(enumerate(texts), key=lambda t: len(t[1]))
        batches = []
        for i in range(0, len(indexed), self.batch_size):
            batch = indexed[i : i + self.batch_size]
            batch_texts = [t for _, t in batch]
            original_indices = [idx for idx, _ in batch]
            max_len = min(
                max(len(t.split()) for t in batch_texts) + 10,
                self.max_seq_len,
            )
            inputs = self.tokenizer(
                batch_texts,
                padding="max_length",
                max_length=max_len,
                truncation=True,
                return_tensors="np",
            )
            batches.append((original_indices, inputs))
        return batches

    async def process_batch(self, batch_inputs):
        """Run inference off the event loop."""
        loop = asyncio.get_event_loop()
        try:
            outputs = await loop.run_in_executor(
                None, self.session.run, None, dict(batch_inputs)
            )
            return outputs
        except Exception as exc:
            logger.error("ONNX batch inference failed: %s", exc)
            return None
```

### Chunking with Entity-Boundary Awareness

```python
from nltk.tokenize import sent_tokenize


def chunk_document(text, max_tokens=512, overlap_tokens=64):
    """Sentence-boundary-aware chunking with token overlap."""
    sentences = sent_tokenize(text)
    chunks, current, length = [], [], 0

    for sentence in sentences:
        n = len(sentence.split())
        if length + n > max_tokens and current:
            chunks.append(" ".join(current))
            # overlap: keep trailing sentences that fit within overlap budget
            tail, tail_len = [], 0
            for s in reversed(current):
                sn = len(s.split())
                if tail_len + sn > overlap_tokens:
                    break
                tail.insert(0, s)
                tail_len += sn
            current = tail + [sentence]
            length = sum(len(s.split()) for s in current)
        else:
            current.append(sentence)
            length += n

    if current:
        chunks.append(" ".join(current))
    return chunks
```

### Worker Pool

```python
import logging
import multiprocessing as mp
from concurrent.futures import ProcessPoolExecutor, as_completed

logger = logging.getLogger(__name__)


class ExtractionWorkerPool:
    def __init__(self, num_workers=None):
        self.num_workers = num_workers or max(mp.cpu_count() - 1, 1)

    def process_batch(self, html_documents):
        chunk_size = max(len(html_documents) // self.num_workers, 1)
        chunks = [
            html_documents[i : i + chunk_size]
            for i in range(0, len(html_documents), chunk_size)
        ]
        results = []
        with ProcessPoolExecutor(max_workers=self.num_workers) as pool:
            futures = {
                pool.submit(_process_chunk, chunk): idx
                for idx, chunk in enumerate(chunks)
            }
            for future in as_completed(futures):
                try:
                    results.extend(future.result())
                except Exception as exc:
                    logger.error("Worker chunk %d failed: %s",
                                 futures[future], exc)
        return results
```

# Novel Insights -- Module 2: Extraction

## Surprising Findings

- **GLiNER's zero-shot NER closes the gap to within 3.8% F1 of fine-tuned BERT (88.5% vs 92.3%) while using 36% less memory and running 15% faster.** The conventional wisdom that you must fine-tune to get usable NER is wrong for B2B text. The cost of maintaining a fine-tuned model (retraining pipeline, annotation queue, versioned weights) dwarfs the marginal F1 gain. GLiNER makes the NER model a config file instead of a training artifact.

- **Entity type *descriptions* beat entity type *labels* by up to 16% F1 on unseen types (ZeroNER, Cocchieri 2025).** Passing "Organization: A company, institution, or business entity" to the model instead of just "ORG" changes the game for zero-shot. This is unintuitive -- you would expect the model to already know what "ORG" means. It does not. The description acts as a soft prompt that disambiguates boundary cases (is "Apple" a fruit or a company?), and the effect is strongest on exactly the entity types you care about most: novel, domain-specific ones like `funding_amount` or `technology`.

- **Joint NER+RE (OneRel) beats pipeline NER-then-RE by 15-20% F1, not because of better models, but because cascading errors compound multiplicatively.** If NER has 92% recall and RE has 85% precision, your end-to-end precision is ~78%. OneRel sidesteps this by treating the problem as a single relational matrix prediction. The latency also drops 37% (58ms vs 92ms) because you eliminate a full second forward pass.

- **Fine-tuned BERT still beats GPT-4 on structured NER by 7-22% F1 while being 100-1000x cheaper per document.** The LLM advantage only appears for implicit relationship inference ("reading between the lines"). For explicit span extraction, smaller supervised models dominate. This contradicts the "just use GPT-4 for everything" narrative.

- **92% of model errors concentrate in the top 20% most uncertain examples (by MC Dropout entropy).** Active learning with uncertainty sampling yields 2.1% F1 per 100 annotations vs 0.8% with random sampling -- a 2.6x efficiency multiplier. This means the naive strategy of "annotate more data" is wasting 60% of annotation budget on examples the model already handles correctly.

- **Softmax confidence from fine-tuned BERT is systematically miscalibrated on out-of-domain text.** The pipeline's 0.75/0.60 thresholds were tuned on validation data but have not been verified with temperature scaling. Typical learned temperature values for fine-tuned BERT fall in [1.2, 2.5], meaning raw softmax probabilities are overconfident by a factor of 1.2-2.5x. A single scalar parameter (temperature) fit on 500 held-out docs fixes this, yet almost nobody does it.

## Unconventional Techniques

- **NER as span-matching against natural language queries (GLiNER architecture).** Instead of token classification with a CRF head, GLiNER frames NER as a bi-encoder matching problem: encode the text, encode the entity type description, compute span-level similarity. This means you can add new entity types at inference time by just writing a new description string. No retraining, no new classification head, no label indexing. The model is schema-driven, not weight-driven.

- **Extraction as relational matrix completion (OneRel).** Instead of NER -> pair enumeration -> relation classification, OneRel predicts a single 2D matrix where rows and columns are token positions and cell values encode both entity boundaries and relation types simultaneously. This reframes IE as a structured prediction problem over a grid rather than a sequence labeling + classification pipeline. It eliminates the O(n^2) pair enumeration bottleneck and makes entity-relation interactions first-class.

- **GritLM: a single model that does both embedding and generation.** GritLM 7B achieves SOTA on MTEB (embedding benchmark) while also being a generative LLM. For Scrapus, this means one model could replace both the sentence-transformer (all-MiniLM-L6-v2 for ChromaDB embeddings) and the small LLM fallback (Phi-3 for complex inference). The architectural trick: instruction-tuned representations that switch behavior based on the prompt prefix. One set of weights, two modalities.

- **Agentic query decomposition for extraction (DocETL, Shankar 2024).** Instead of running a fixed extraction pipeline, DocETL uses a planning agent to decompose complex extraction queries into simpler sub-tasks, routes each to specialized extractors, then validates and synthesizes results. The counterintuitive insight: adding an LLM planning layer *on top of* specialized models improves accuracy more than replacing the specialized models with a bigger LLM.

- **Visual NER from rendered HTML (Jung et al., 2022 -- First Impression Area).** Render the webpage, use the visual layout as a feature for extraction. This works especially well for non-English pages where text-based boilerplate removal fails. The "first impression area" heuristic (the region users see first on page load) outperforms DOM-tree traversal for content extraction on visually complex pages. Nobody in the NER community talks about this because they only look at clean text, but real web pages are messy visual artifacts.

- **Dependency-path pruning at 4 hops kills long-distance relations, but increasing the hop limit doesn't help -- it just adds noise.** The better fix is to switch to a graph transformer (Chaturvedi et al., 2025) that learns attention over the dependency graph rather than hard-coding a hop limit. This gives 12% improvement on complex relations while handling cross-clause references that the 4-hop window misses.

## Hidden Gems (Papers)

- **"A Frustratingly Easy Approach for Entity and Relation Extraction" (Zhong & Chen, 2021).** Despite the tongue-in-cheek title, this paper shows that using *independent encoders* for NER and RE (rather than shared encoders) matches or beats joint models. The insight: shared representations create optimization conflicts between the two tasks. The Scrapus pipeline accidentally does this right by using separate BERT models for NER and RE, but for the wrong reason (they were just built independently).

- **Few-NERD (Ding et al., 2021).** A dataset specifically for few-shot NER with fine-grained entity types. Critical for evaluating how well GLiNER or ZeroNER would perform on the long-tail B2B entity types (technology names, funding instruments, industry verticals) that CoNLL-2003 does not cover. Most people evaluate zero-shot NER on CoNLL -- this is cheating because the models have seen CoNLL during pretraining.

- **Template-based NER using BART (Cui et al., 2021).** Reframes NER as a seq2seq problem: given "Acme Corp launched AI ThreatGuard", generate "Acme Corp is an organization, AI ThreatGuard is a product". Weird, but it works for few-shot scenarios where you have <50 examples. The decoder's autoregressive nature handles overlapping entities naturally, which BIO tagging cannot.

- **Stepanov & Shtopko (2024) -- GLiNER multi-task.** Shows 15-25% improvement over BERT-base in few-shot scenarios by extending GLiNER to handle NER, text classification, and relation extraction in a single forward pass. Underappreciated because it was published as an arXiv preprint, not a top venue paper.

- **Word-word relation classification for NER (Li et al., 2022).** Handles flat, overlapped, and discontinuous entities by predicting relations between every pair of words in the sentence. This is the only approach that correctly handles discontinuous entities like "New ... York" separated by other tokens. Relevant for messy web text where entity mentions are interrupted by HTML artifacts or parentheticals.

## Contrarian Takes

- **The 512-token chunk with 64-token overlap is a cargo-cult default that actively hurts extraction quality.** The overlap ensures some entities appear in multiple chunks, but there is no cross-chunk deduplication or coreference resolution. This means the same entity gets counted multiple times, inflating entity counts and creating phantom duplicates in ChromaDB. Either implement proper cross-chunk merging or stop pretending the overlap helps.

- **BERTopic + LDA dual topic modeling is redundant engineering, not defense in depth.** LDA produces a 20-dim topic vector. BERTopic produces a topic ID + key phrases from the same embeddings used for ChromaDB storage. The LDA vector adds almost zero information over the embedding itself -- any linear probe on the 384-dim embedding will recover the LDA topics. The only justification is interpretability, but BERTopic's c-TF-IDF phrases are more interpretable than LDA topic IDs anyway. Drop LDA.

- **The reward signal to the crawler (1.0 / 0.2 / -0.1) is a blunt instrument that creates a degenerate feedback loop.** Pages with lots of ORG entities get reward=1.0, which biases the crawler toward pages that mention many companies (press release aggregators, company directories) rather than pages with *useful information about specific companies*. A reward based on entity *novelty* (new entities not already in ChromaDB) would be far more valuable than entity *count*.

- **Confidence thresholds per entity type (0.75 for ORG/PERSON, 0.60 for LOCATION/PRODUCT) solve the wrong problem.** The real issue is that the model's softmax scores are uncalibrated. Lowering the threshold for PRODUCT compensates for lower training data, but it also admits more false positives. Temperature scaling with a single learned parameter would fix calibration across all types simultaneously. The per-type threshold is a manual patch over a systematic calibration failure.

- **ONNX INT8 quantization promises 3x speedup with <0.5% F1 loss, but this only holds for encoder models.** For the SpERT/OneRel joint extraction models that have complex span enumeration and pairwise relation heads, quantization of the classification layers (not just the encoder) causes larger accuracy drops (1-3% F1). The encoder quantizes well; the task-specific heads do not. Selective quantization (encoder INT8, heads FP16) is the correct approach but is not supported out-of-the-box by `onnxruntime.quantization`.

## Wild Ideas

- **Entity discovery via embedding clustering instead of NER.** Skip NER entirely. Embed every noun phrase in the document using a sentence transformer, cluster them with HDBSCAN, and label clusters post-hoc based on their centroid's nearest neighbor in a type-description embedding space. This inverts the NER paradigm: instead of "find spans that match predefined types," you "find spans that form natural groups and then name the groups." Advantage: discovers entity types you didn't know to look for. A B2B page might contain `board_advisor` or `patent_filing` entities that no predefined schema includes.

- **Multimodal extraction from webpage screenshots.** Render the page as a PNG, run a vision-language model (LLaVA, Qwen-VL) with the prompt "extract all company names, products, and business relationships visible on this page." This sounds insane but it handles: (1) text embedded in images/logos, (2) table structures that DOM parsing mangles, (3) visual proximity cues that indicate relationships ("this person's photo is next to this company's logo"). Current web extractors throw away all visual signal.

- **Extraction as code generation.** Instead of structured output from an NER model, prompt a code-generation LLM to write Python code that extracts the entities from the raw text. The LLM generates `entities = [{"name": "Acme Corp", "type": "ORG"}, ...]` as executable code. Execute it in a sandbox. Advantage: the LLM can express complex extraction logic (regex for funding amounts, date parsing, conditional entity typing) that no fixed classification head can represent. This is essentially program synthesis for IE.

- **Cross-document entity graphs via embedding-space walks.** After extracting entities from individual pages, don't just store them -- embed them and build a nearest-neighbor graph in embedding space. Walk the graph to discover implicit relationships: if "Acme Corp" is close to "BetaCorp" in embedding space across many pages, they are likely competitors, even if no single page states this explicitly. The graph structure encodes latent business relationships that no single-document extractor can find.

- **Self-supervised NER from anchor text.** Hyperlinks in crawled HTML are free entity annotations: the anchor text is an entity mention, and the URL's domain hints at the entity type (linkedin.com -> PERSON, crunchbase.com -> ORG). Build a silver-standard NER training set from millions of crawled hyperlinks without any human annotation. The noise rate is high (~30%), but the volume compensates. Train a student model on this silver data, then distill with knowledge distillation from a teacher model run on the clean subset.

- **Temporal extraction as event timelines.** Instead of static (subject, predicate, object) triples, extract (subject, predicate, object, timestamp, source_url) quintuples. Use date parsing (dateutil, regex) on the surrounding context of each relation to assign temporal bounds. This turns the knowledge base from a static graph into a temporal event log, enabling queries like "show me all acquisitions in cybersecurity in Q1 2025" that the current pipeline cannot answer. The relation extraction model does not need to change -- just add a date extraction head that fires on each relation's context window.

```python
import torch
import torch.nn as nn
from transformers import AutoModel, AutoConfig

class SpERT_Scrapus(nn.Module):
    """Span-based joint NER+RE for B2B lead generation"""
    
    def __init__(self, model_name="microsoft/deberta-v3-base"):
        super().__init__()
        self.encoder = AutoModel.from_pretrained(model_name)
        config = AutoConfig.from_pretrained(model_name)
        hidden_size = config.hidden_size
        
        # Span representation layers
        self.span_start = nn.Linear(hidden_size, hidden_size)
        self.span_end = nn.Linear(hidden_size, hidden_size)
        self.span_width = nn.Embedding(50, hidden_size)  # Max span length
        
        # Entity classification head
        self.entity_classifier = nn.Linear(hidden_size * 3, len(ENTITY_TYPES))
        
        # Relation classification head
        self.relation_classifier = nn.Sequential(
            nn.Linear(hidden_size * 6, hidden_size),
            nn.ReLU(),
            nn.Dropout(0.1),
            nn.Linear(hidden_size, len(RELATION_TYPES))
        )
    
    def forward(self, input_ids, attention_mask, spans):
        # Get contextual embeddings
        outputs = self.encoder(input_ids, attention_mask=attention_mask)
        sequence_output = outputs.last_hidden_state
        
        # Extract span representations
        span_reps = []
        for (start, end) in spans:
            start_rep = self.span_start(sequence_output[:, start])
            end_rep = self.span_end(sequence_output[:, end])
            width_rep = self.span_width(end - start)
            
            # Concatenate representations
            span_rep = torch.cat([start_rep, end_rep, width_rep], dim=-1)
            span_reps.append(span_rep)
        
        span_reps = torch.stack(span_reps, dim=1)
        
        # Entity classification
        entity_logits = self.entity_classifier(span_reps)
        
        # Relation classification (pairwise)
        relation_logits = []
        for i in range(len(spans)):
            for j in range(len(spans)):
                if i != j:
                    pair_rep = torch.cat([span_reps[:, i], span_reps[:, j]], dim=-1)
                    rel_logit = self.relation_classifier(pair_rep)
                    relation_logits.append(rel_logit)
        
        return entity_logits, torch.stack(relation_logits, dim=1)
```

```python
    def _compute_entropy(self, predictions, task):
        """Compute predictive entropy"""
        all_probs = []
        for pred in predictions:
            if task == "entities":
                probs = torch.softmax(pred[task], dim=-1)
            else:
                probs = torch.softmax(pred[task], dim=-1)
            all_probs.append(probs)
        
        avg_probs = torch.mean(torch.stack(all_probs), dim=0)
        entropy = -torch.sum(avg_probs * torch.log(avg_probs + 1e-10))
        return entropy.item()
```
