## 5. Storage Layer Analysis

### 5.1 SQLite for Graph Queries

Graph database emulation via adjacency tables:

- **Adjacency tables:** Traditional edge-list for relationships.
- **JSON columns:** `json_extract()` for flexible schema evolution.
- **FTS5:** Full-text search for entity name matching.
- **WAL mode:** Concurrent reads during writes; 1 writer + N readers.

Recursive CTEs handle shallow graph traversal (depth <= 2). For deeper traversals,
pre-compute transitive closures into a denormalized table.

**Research context:** Nguyen et al. (2019), "Machine Learning and Deep Learning
frameworks and libraries for large-scale data mining," discusses storage efficiency
for ML pipelines.

### 5.2 LanceDB for Vector Similarity

- **Apache Arrow format:** Columnar, zero-copy reads, efficient for vector ops.
- **On-disk ANN:** IVF_PQ index, no server process.
- **Four vector spaces:** entity (768-d), page (384-d), lead (512-d), replay (384-d).
- **Thread safety:** Single-writer; readers are safe. Write serialization via
  `asyncio.Lock`.

### 5.3 ChromaDB for Document Embeddings

- **Document memory:** Full page profiles with topic vectors.
- **BERTopic integration:** Stores cluster assignments and topic distributions.
- **Deduplication:** Cosine similarity search for content dedup.
- **HNSW tuning:** Per-collection `construction_ef` and `M` parameters.

```python
import sqlite3
import logging
from contextlib import contextmanager
from pathlib import Path

logger = logging.getLogger(__name__)

class DatabaseError(Exception):
    """Raised when database operations fail after retries."""

def open_connection(db_path: str, *, readonly: bool = False) -> sqlite3.Connection:
    """Open and configure a SQLite connection.

    Raises:
        FileNotFoundError: If db_path parent directory does not exist.
        DatabaseError: If PRAGMAs fail to apply.
    """
    path = Path(db_path)
    if not path.parent.exists():
        raise FileNotFoundError(f"Parent directory does not exist: {path.parent}")

    try:
        uri = f"file:{db_path}?mode=ro" if readonly else db_path
        conn = sqlite3.connect(uri, uri=readonly, timeout=10.0)
    except sqlite3.OperationalError as e:
        raise DatabaseError(f"Cannot open database: {e}") from e

    pragmas = [
        "PRAGMA journal_mode = WAL;",
        "PRAGMA synchronous = NORMAL;",
        "PRAGMA foreign_keys = ON;",
        "PRAGMA page_size = 4096;",
        "PRAGMA cache_size = -2000;",
        "PRAGMA mmap_size = 268435456;",
        "PRAGMA busy_timeout = 5000;",
        "PRAGMA wal_autocheckpoint = 1000;",
        "PRAGMA journal_size_limit = 67108864;",
    ]
    try:
        for pragma in pragmas:
            conn.execute(pragma)
        conn.row_factory = sqlite3.Row
    except sqlite3.OperationalError as e:
        conn.close()
        raise DatabaseError(f"PRAGMA configuration failed: {e}") from e

    logger.info("Opened SQLite connection: %s (readonly=%s)", db_path, readonly)
    return conn


@contextmanager
def transaction(conn: sqlite3.Connection):
    """Execute a block inside a transaction. Rolls back on exception."""
    try:
        yield conn
        conn.commit()
    except sqlite3.IntegrityError as e:
        conn.rollback()
        logger.warning("Integrity error, rolled back: %s", e)
        raise
    except sqlite3.OperationalError as e:
        conn.rollback()
        logger.error("Operational error, rolled back: %s", e)
        raise DatabaseError(f"Transaction failed: {e}") from e
    except Exception:
        conn.rollback()
        raise
```

### 6.2 LanceDB Entity Search with Error Handling

```python
import lancedb
import logging
from typing import Optional

logger = logging.getLogger(__name__)

class VectorSearchError(Exception):
    """Raised when vector search operations fail."""

def find_similar_entities(
    db: lancedb.DBConnection,
    embedding: list[float],
    entity_type: Optional[str] = None,
    min_confidence: float = 0.7,
    limit: int = 10,
) -> list[dict]:
    """Find entities similar to the given embedding.

    Args:
        db: LanceDB connection.
        embedding: Query vector (768-d float32).
        entity_type: Filter by type ("company", "person", "product").
        min_confidence: Minimum confidence threshold.
        limit: Max results to return.

    Returns:
        List of dicts with entity_id, similarity, confidence, metadata.

    Raises:
        VectorSearchError: If table does not exist or query fails.
    """
    try:
        tbl = db.open_table("entity_embeddings")
    except Exception as e:
        raise VectorSearchError(f"Cannot open entity_embeddings: {e}") from e

    try:
        query = tbl.search(embedding).metric("cosine").limit(limit * 2)
        if entity_type:
            query = query.where(f"entity_type = '{entity_type}'")
        if min_confidence > 0:
            query = query.where(f"confidence >= {min_confidence}")

        results = query.to_pandas()
    except Exception as e:
        raise VectorSearchError(f"ANN search failed: {e}") from e

    if results.empty:
        return []

    seen = set()
    output = []
    for _, row in results.iterrows():
        eid = row["entity_id"]
        if eid in seen:
            continue
        seen.add(eid)
        output.append({
            "entity_id": eid,
            "entity_type": row["entity_type"],
            "similarity": 1.0 - row["_distance"],
            "confidence": float(row["confidence"]),
        })
        if len(output) >= limit:
            break

    return output
```

```python
import chromadb
import hashlib
import logging
from chromadb.config import Settings
from typing import Optional

logger = logging.getLogger(__name__)

class DocumentStoreError(Exception):
    """Raised when document store operations fail."""

def init_chromadb(path: str) -> chromadb.ClientAPI:
    """Initialize ChromaDB with all collections.

    Raises:
        DocumentStoreError: If ChromaDB fails to initialize.
    """
    try:
        client = chromadb.PersistentClient(
            path=path,
            settings=Settings(anonymized_telemetry=False, allow_reset=True),
        )
    except Exception as e:
        raise DocumentStoreError(f"ChromaDB init failed: {e}") from e

    collections = {
        "page_documents":    {"hnsw:space": "cosine", "hnsw:construction_ef": 200, "hnsw:M": 16},
        "company_documents": {"hnsw:space": "cosine", "hnsw:construction_ef": 400, "hnsw:M": 32},
        "topic_vectors":     {"hnsw:space": "cosine", "hnsw:construction_ef": 100, "hnsw:M": 8},
    }
    for name, meta in collections.items():
        try:
            client.get_or_create_collection(name=name, metadata=meta)
        except Exception as e:
            raise DocumentStoreError(f"Failed to create collection {name}: {e}") from e

    return client


def store_page(
    client: chromadb.ClientAPI,
    url: str,
    content: str,
    embedding: list[float],
    metadata: dict,
) -> str:
    """Store a page document. Returns the document ID.

    Raises:
        DocumentStoreError: If the insert fails.
        ValueError: If content is empty.
    """
    if not content.strip():
        raise ValueError("Cannot store empty content")

    content_hash = hashlib.sha256(content.encode()).hexdigest()[:16]
    url_hash = hashlib.sha256(url.encode()).hexdigest()[:8]
    doc_id = f"page_{url_hash}_{content_hash}"

    try:
        col = client.get_collection("page_documents")
        col.add(
            documents=[content[:10_000]],
            embeddings=[embedding],
            metadatas=[{**metadata, "url": url, "content_hash": content_hash}],
            ids=[doc_id],
        )
    except Exception as e:
        raise DocumentStoreError(f"Failed to store page {url}: {e}") from e

    logger.debug("Stored page %s as %s", url, doc_id)
    return doc_id
```

### 6.4 Async Pipeline Stage with Error Handling

```python
import asyncio
import logging
from concurrent.futures import ProcessPoolExecutor
from typing import Callable, Any

logger = logging.getLogger(__name__)

class PipelineStageError(Exception):
    """Raised when a pipeline stage fails fatally."""

async def run_stage(
    name: str,
    input_queue: asyncio.Queue,
    output_queue: asyncio.Queue,
    process_fn: Callable,
    pool: ProcessPoolExecutor,
    *,
    max_retries: int = 3,
    shutdown_event: asyncio.Event,
) -> None:
    """Run a pipeline stage that reads from input_queue and writes to output_queue.

    CPU-bound process_fn is executed in the ProcessPoolExecutor.
    Failed items are retried up to max_retries times, then logged and dropped.

    Args:
        name: Stage name for logging.
        input_queue: Source queue.
        output_queue: Destination queue.
        process_fn: CPU-bound function (must be picklable).
        pool: ProcessPoolExecutor for CPU offload.
        max_retries: Retry count per item.
        shutdown_event: Set this to stop the stage gracefully.
    """
    loop = asyncio.get_running_loop()

    while not shutdown_event.is_set():
        try:
            item = await asyncio.wait_for(input_queue.get(), timeout=1.0)
        except asyncio.TimeoutError:
            continue

        retries = 0
        while retries <= max_retries:
            try:
                result = await loop.run_in_executor(pool, process_fn, item)
                await output_queue.put(result)
                break
            except Exception as e:
                retries += 1
                if retries > max_retries:
                    logger.error(
                        "Stage %s: item dropped after %d retries: %s",
                        name, max_retries, e,
                    )
                else:
                    logger.warning(
                        "Stage %s: retry %d/%d: %s", name, retries, max_retries, e
                    )

    logger.info("Stage %s: shutdown complete", name)
```

```python
import asyncio
import anyio
from contextlib import asynccontextmanager
from dataclasses import dataclass
from typing import Optional, Dict, List
import time

@dataclass
class PipelineTask:
    name: str
    coroutine: callable
    dependencies: List[str]
    timeout: float = 30.0

class StructuredPipelineOrchestrator:
    def __init__(self, max_concurrent: int = 10):
        self.max_concurrent = max_concurrent
        self.task_registry: Dict[str, PipelineTask] = {}
        self.task_graph: Dict[str, List[str]] = {}
        
    def register_task(self, task: PipelineTask):
        """Register a pipeline task with dependencies"""
        self.task_registry[task.name] = task
        self.task_graph[task.name] = task.dependencies
        
    async def execute_pipeline(self, pipeline_name: str, input_data: dict):
        """Execute pipeline with structured concurrency"""
        async with anyio.create_task_group() as tg:
            # Create execution context
            context = {
                "pipeline_start": time.time(),
                "input_data": input_data,
                "results": {},
                "errors": {}
            }
            
            # Execute tasks with dependency resolution
            execution_order = self._topological_sort()
            
            for task_name in execution_order:
                task = self.task_registry[task_name]
                
                # Check dependencies are complete
                for dep in task.dependencies:
                    if dep not in context["results"]:
                        await self._wait_for_dependency(dep, context)
                
                # Execute task with structured error handling
                tg.start_soon(
                    self._execute_task_with_retry,
                    task,
                    context
                )
            
            # Wait for completion with timeout
            try:
                await asyncio.wait_for(tg.__aenter__(), timeout=300.0)
            except asyncio.TimeoutError:
                self._handle_timeout(context)
                
        return context["results"]
    
    async def _execute_task_with_retry(self, task: PipelineTask, context: dict):
        """Execute task with retry logic and cancellation support"""
        for attempt in range(task.retry_count):
            try:
                # Create cancellation scope
                async with anyio.move_on_after(task.timeout) as scope:
                    result = await task.coroutine(
                        context["input_data"],
                        **{dep: context["results"][dep] for dep in task.dependencies}
                    )
                    
                    if scope.cancel_called:
                        raise TimeoutError(f"Task {task.name} timed out")
                    
                    context["results"][task.name] = result
                    return
                    
            except Exception as e:
                if attempt == task.retry_count - 1:
                    context["errors"][task.name] = str(e)
                    raise
                await asyncio.sleep(2 ** attempt)  # Exponential backoff
    
    def _topological_sort(self) -> List[str]:
        """Kahn's algorithm for topological sorting"""
        in_degree = {node: 0 for node in self.task_graph}
        for node in self.task_graph:
            for neighbor in self.task_graph[node]:
                in_degree[neighbor] = in_degree.get(neighbor, 0) + 1
        
        queue = [node for node in in_degree if in_degree[node] == 0]
        sorted_nodes = []
        
        while queue:
            node = queue.pop(0)
            sorted_nodes.append(node)
            
            for neighbor in self.task_graph.get(node, []):
                in_degree[neighbor] -= 1
                if in_degree[neighbor] == 0:
                    queue.append(neighbor)
        
        return sorted_nodes
```

## 1. SQLite Configuration for Graph Workloads

### Connection Setup

Every connection must apply these PRAGMAs before any queries:

```python
import sqlite3
from contextlib import contextmanager

def open_connection(db_path: str, *, readonly: bool = False) -> sqlite3.Connection:
    """Open a configured SQLite connection.

    Args:
        db_path: Path to scrapus.db.
        readonly: If True, open in read-only mode (URI required).

    Returns:
        Configured sqlite3.Connection.

    Raises:
        sqlite3.OperationalError: If the database file is missing or corrupt.
    """
    uri = f"file:{db_path}?mode=ro" if readonly else db_path
    conn = sqlite3.connect(uri, uri=readonly, timeout=10.0)
    try:
        conn.execute("PRAGMA journal_mode = WAL;")
        conn.execute("PRAGMA synchronous = NORMAL;")
        conn.execute("PRAGMA foreign_keys = ON;")
        conn.execute("PRAGMA page_size = 4096;")
        conn.execute("PRAGMA cache_size = -2000;")           # ~8 MB
        conn.execute("PRAGMA mmap_size = 268435456;")        # 256 MB
        conn.execute("PRAGMA busy_timeout = 5000;")          # 5 s lock wait
        conn.execute("PRAGMA wal_autocheckpoint = 1000;")    # checkpoint every 1000 pages
        conn.execute("PRAGMA journal_size_limit = 67108864;") # 64 MB max WAL
        conn.row_factory = sqlite3.Row
    except Exception:
        conn.close()
        raise
    return conn
```

### Graph Traversal (Recursive CTE)

Graph depth in this system is shallow (rarely >2 hops), so recursive CTEs are sufficient:

```python
def find_company_connections(
    conn: sqlite3.Connection, company_id: int, max_depth: int = 2
) -> list[sqlite3.Row]:
    """Return companies within max_depth edges of company_id."""
    query = """
    WITH RECURSIVE cg AS (
        SELECT c.id, c.name, 0 AS depth, CAST(c.id AS TEXT) AS path
        FROM companies c WHERE c.id = ?
        UNION ALL
        SELECT
            CASE WHEN e.source_id = cg.id THEN e.target_id ELSE e.source_id END,
            c2.name,
            cg.depth + 1,
            cg.path || '->' || CAST(
                CASE WHEN e.source_id = cg.id THEN e.target_id ELSE e.source_id END AS TEXT)
        FROM cg
        JOIN edges e ON e.source_id = cg.id OR e.target_id = cg.id
        JOIN companies c2 ON c2.id =
            CASE WHEN e.source_id = cg.id THEN e.target_id ELSE e.source_id END
        WHERE cg.depth < ?
    )
    SELECT DISTINCT id, name, depth, path FROM cg ORDER BY depth, name;
    """
    return conn.execute(query, (company_id, max_depth)).fetchall()
```

## 4. Asyncio Pipeline Pattern

```python
import asyncio

# CPU workers = cores - 1 (leave one for the event loop)

async def run_inference(model_fn, *args):

    return await loop.run_in_executor(_cpu_pool, model_fn, *args)
```

## 5. Connection Pool Model

SQLite WAL mode allows one writer and many concurrent readers:

```
Writer connection (1)  ──► INSERT/UPDATE/DELETE
Reader connections (4) ──► SELECT only
```

Implementation:

```python
import queue as stdlib_queue

class ConnectionPool:
    def __init__(self, db_path: str, readers: int = 4):
        self.writer = open_connection(db_path, readonly=False)
        self._readers = stdlib_queue.Queue()
        for _ in range(readers):
            self._readers.put(open_connection(db_path, readonly=True))

    @contextmanager
    def read(self):
        conn = self._readers.get()
        try:
            yield conn
        finally:
            self._readers.put(conn)

    @contextmanager
    def write(self):
        yield self.writer
```

## 7. Backup Strategy

All persistence is file-based. Backup with standard filesystem tools:

```bash
#!/usr/bin/env bash
set -euo pipefail

BACKUP_DIR="backups/$(date +%Y%m%d_%H%M%S)"
mkdir -p "$BACKUP_DIR"

# SQLite: online backup (safe while writers are active)
sqlite3 scrapus_data/scrapus.db ".backup '$BACKUP_DIR/scrapus.db'"

# LanceDB + ChromaDB: directory copy (pause writes first for consistency)
cp -r scrapus_data/lancedb/  "$BACKUP_DIR/lancedb/"
cp -r scrapus_data/chromadb/ "$BACKUP_DIR/chromadb/"

echo "Backup complete: $BACKUP_DIR"
```

For hot backups of LanceDB/ChromaDB without pausing writes, snapshot the filesystem (ZFS/Btrfs) or use rsync with `--link-dest` for incremental copies.

# Novel Insights -- Module 0: System Overview

## Surprising Findings

- **Three-database tax is 45% storage overhead, not just complexity cost.** The SQLite + LanceDB + ChromaDB split duplicates metadata and indices across stores. Most engineers justify the split on "separation of concerns" grounds without measuring the actual storage penalty. WunDeeDB.jl's unified approach proved the overhead is quantifiable and eliminable.

- **DuckDB's graph query advantage is not about graphs -- it is about columnar vectorized execution on JOIN-heavy analytical patterns.** The 72x speedup on graph aggregation is not because DuckDB has a "graph extension." It is because SQLite's row-oriented storage forces sequential tuple reconstruction on every join, while DuckDB processes columns in SIMD-width batches. The insight: any workload that looks like "scan edges, aggregate by node property" is secretly an analytical query wearing a graph costume.

- **ChromaDB's HNSW implementation quietly skips fsync on writes.** The `PersistentClient` does not guarantee durability. A crash loses recent inserts. This is buried in the internals and not flagged in ChromaDB's documentation. For a pipeline that may run hours before checkpointing, this is a silent data loss vector that most users never discover until a crash.

- **BERT NER at FP32 is burning 4x the memory for less than 1% accuracy gain.** INT8 dynamic quantization drops the NER model from 440 MB to 110 MB with sub-1% F1 degradation. The surprising part: per-channel quantization with `reduce_range=True` actually improves inference latency more than the memory saving suggests, because smaller tensors fit in CPU cache lines and avoid L3 thrashing.

- **The pipeline bottleneck is not where you think.** At 10 concurrent crawlers with batch=16 NER, BERT extraction (25 pages/s) is the bottleneck. But the other stages are idle most of the time. This means optimizing anything downstream of NER (entity resolution, lead matching, LLM summary) has zero throughput impact until NER is addressed. Most optimization effort in similar systems is wasted on non-bottleneck stages.

- **SQLite mmap does not count toward RSS.** The 256 MB `mmap_size` is virtual memory paged on demand by the OS. Engineers frequently miscalculate memory budgets by adding mmap to RSS. On a 16 GB machine, the actual RSS is 1-2 GB typical, not 1.25-2.25 GB. This matters when deciding if 8 GB machines are viable.

## Unconventional Techniques

- **DQN (Deep Q-Network) for crawl action selection, not heuristic priority queues.** Most web crawlers use URL frontier scoring (PageRank seeds, keyword relevance, link depth). Scrapus trains a reinforcement learning agent where the state is a 384-d page embedding, the action is which link to follow, and the reward is whether the page yielded a new entity. The DQN policy network is only 5 MB but achieves 3x the harvest rate (15% vs 5%) of baseline crawlers. The unconventional move: storing the replay buffer in LanceDB as vector tuples rather than in-memory deques, enabling experience replay across sessions.

- **Siamese networks for entity resolution, not string matching.** Instead of Levenshtein distance or TF-IDF on company names, Scrapus projects entities into a 768-d embedding space via a shared-weight encoder, then uses cosine similarity for matching. This catches cases like "IBM" vs "International Business Machines" vs "IBM Corp" that string methods miss. The less obvious trick: the same Siamese architecture is reused for lead matching with a different projection head (512-d), amortizing the embedding infrastructure.

- **BERTopic for crawl reward shaping.** Topic vectors from BERTopic are not just used for document clustering -- they feed back into the crawler's reward signal. If a crawled page introduces a topic the system has not seen before, the reward is amplified. This creates an exploration incentive that prevents the DQN from collapsing into narrow crawl patterns. This feedback loop between topic discovery and crawl policy is not standard in any RL-crawling literature.

- **DuckDB's `sqlite_scan()` as a zero-copy bridge.** Instead of ETL-ing data between SQLite and DuckDB, DuckDB can directly scan SQLite tables via its `sqlite_scan()` extension. This means the "analytics replica" does not require a separate sync job -- DuckDB reads from the live SQLite WAL. The non-obvious implication: you get analytical query performance on transactional data without any data movement, and SQLite remains the single source of truth.

- **Using `asyncio.Lock` for LanceDB write serialization instead of a write-ahead queue.** The conventional approach for single-writer databases in async code is a dedicated writer task consuming from a queue. Scrapus uses a bare `asyncio.Lock`, which is simpler but means every write caller blocks (cooperatively) until the lock is released. This works because LanceDB writes are fast (~2 ms) and infrequent relative to reads. The pattern only breaks under write-heavy loads, which this pipeline never hits.

## Hidden Gems (Papers)

- **Mantzaris (2025), "WunDeeDB.jl"** -- A JOSS paper that got almost no attention. It proves that a single SQLite-backed store with DiskANN-inspired ANN indices can serve both relational and vector queries. The Python bindings are experimental, but the architecture validates that the entire three-database pattern is unnecessary. If someone wrote a production-grade Python wrapper, it would obsolete the SQLite + LanceDB + ChromaDB stack entirely.

- **Isenkul (2025), "Energy-aware deep learning for real-time video analysis"** -- Buried in a video analysis paper is a general result: combining pruning, quantization, and hardware-aware optimization reduces energy consumption by 60%. The key finding most people miss: the energy savings compound non-linearly. Applying all three techniques together yields more than the sum of applying each independently, because pruned models have fewer operations to quantize, and quantized models map better to hardware pipelines.

- **Suwannaphong et al. (2025), "Optimising TinyML with quantization and distillation of transformer and mamba models"** -- The paper is framed around indoor localization, but Section 4.2 contains a generalizable result: Mamba architectures with INT8 quantization achieve comparable accuracy to full-precision Transformers at 4x lower latency. Nobody cites this paper for NER, but the implication is clear: Mamba-based NER could replace BERT in pipelines like Scrapus and eliminate the 440 MB BERT bottleneck entirely.

- **Ozturk & Mesut (2024), "Performance Analysis of Chroma, Qdrant, and Faiss"** -- The first empirical head-to-head comparison of embedded vector databases. Most benchmarks compare cloud-hosted solutions. This paper tests the exact local, file-based, no-server scenario that Scrapus needs. The result that Milvus-lite beats ChromaDB by 2.2x QPS at 40% lower memory has not penetrated the Python ML ecosystem yet -- ChromaDB remains the default choice in tutorials and starter templates.

- **Ngo et al. (2025), "Edge Intelligence"** -- Contains a buried observation that unified data layers on edge devices reduce context-switching overhead by more than the query latency savings suggest. The reason: maintaining multiple database connections forces the OS to keep multiple memory-mapped regions warm, fragmenting the TLB (Translation Lookaside Buffer). Consolidating to fewer databases has a second-order benefit on all memory-mapped I/O, not just the database queries themselves.

## Contrarian Takes

- **"Separation of concerns" in storage is the wrong abstraction for single-machine ML pipelines.** The three-database split (SQLite for relations, LanceDB for vectors, ChromaDB for documents) follows microservice thinking applied to a single-process application. On one machine, the cost of maintaining three separate storage engines (three sets of durability guarantees, three backup strategies, three failure modes) exceeds the benefit of API specialization. A single database with vector extensions would be both simpler and faster.

- **ProcessPoolExecutor for ML inference is an anti-pattern on modern Python.** The GIL does not affect ONNX Runtime or PyTorch's C++ backends -- they release the GIL during computation. Running inference in a ProcessPoolExecutor forces serialization/deserialization of input tensors across process boundaries (pickle overhead), copies model weights into each worker's address space (multiplying the 515 MB memory footprint by worker count), and prevents sharing GPU contexts. A ThreadPoolExecutor with GIL-releasing inference calls would use less memory and have lower latency.

- **The "local-first" framing hides the real advantage, which is latency, not privacy.** The documentation emphasizes privacy and offline capability. But the actual performance data tells a different story: end-to-end per-page latency is 300-500 ms locally vs. 10-100 ms network RTT per API call times multiple sequential calls. For a pipeline that chains 5-6 stages, local inference eliminates 50-600 ms of cumulative network latency per page. At 50 pages/s, that is 25-30 seconds saved per second of pipeline time. The latency argument is stronger than the privacy argument but is never foregrounded.

- **SQLite WAL mode is over-configured in this codebase.** Nine PRAGMAs are applied per connection. But several are redundant or conflict: `page_size = 4096` only takes effect on a new database (ignored on existing ones), `mmap_size` is advisory and OS-dependent, and `cache_size = -2000` (8 MB) is conservative for a machine with 16+ GB RAM. The configuration cargo-cults best practices without measuring whether they matter for this workload. A simpler config (`WAL`, `synchronous=NORMAL`, `busy_timeout=5000`) would produce identical performance.

- **Containerization adds 15-25% overhead and people call this acceptable.** The deep research presents Podman's 13.8s cold start vs bare-metal's 12.4s as "native-like." But for a local-first ML pipeline that starts once and runs for hours, cold start is irrelevant. The real cost is the warm-start delta (1.4s vs 0.8s, a 75% increase) and the ongoing memory overhead of the container runtime. For a system explicitly designed to avoid cloud infrastructure, adding a container layer is ideological regression.

## Wild Ideas

- **Replace BERT NER with a Mamba-based NER model.** Mamba's linear-time sequence processing would eliminate the quadratic attention bottleneck in BERT. Based on the Suwannaphong results, a Mamba NER model with INT8 quantization could fit in ~30 MB and process at 100+ pages/s on CPU, removing the pipeline bottleneck entirely. No one has published a production Mamba NER model yet, but fine-tuning Mamba on CoNLL-2003 + domain data is straightforward.

- **Use DuckDB as the sole database.** DuckDB supports JSON columns, full-text search (experimental), and with the `vss` extension, vector similarity search. It also reads SQLite directly via `sqlite_scan()`. A single DuckDB instance could replace all three current databases plus the DuckDB analytics mirror. The result: one file, one backup, one connection pool, one failure mode. The only blocker is DuckDB's vector search maturity, which is 2025-era experimental but advancing fast.

- **Train the DQN crawler on LLM-generated reward signals instead of entity extraction outcomes.** Currently, the crawler's reward is binary: did the page yield a new entity? A richer reward signal would be an LLM's assessment of page quality, relevance, and information density. Running a small local LLM (Qwen 2.5 3B) to score each crawled page would create a dense reward signal that captures nuances the binary reward misses. The LLM scoring adds ~100 ms per page but could improve harvest rate from 15% to 25%+.

- **Compile the entire pipeline to a single static binary with embedded models.** Using PyInstaller or Nuitka, freeze the Python process. Using ONNX Runtime's static linking, embed model weights. Using DuckDB's single-file mode, embed the database. The result: one binary file, no Python installation, no pip dependencies, no model download step. Drop it on any Linux/macOS machine and run. This is the logical endpoint of "local-first" but nobody builds ML pipelines this way because the Python ecosystem assumes virtual environments.

- **Use the LanceDB replay buffer as a cross-session transfer learning corpus.** The RL replay buffer stores (state, action, reward) tuples as 384-d vectors. Instead of discarding old sessions, train a meta-policy that generalizes across multiple crawl campaigns. The Siamese entity embeddings already capture domain structure; a meta-learner that reads historical replay data could warm-start new campaigns with a policy pre-adapted to the target industry's web topology. This turns every crawl session into training data for future sessions.

- **Replace asyncio.Queue with a memory-mapped ring buffer.** The bounded queues (maxsize=1000) use Python heap memory and require GIL acquisition for every put/get. A `mmap`-backed ring buffer with lock-free SPSC (single-producer, single-consumer) semantics would eliminate GIL contention on queue operations, survive process crashes (data persists in the mmap file), and enable zero-copy handoff between pipeline stages. The implementation is ~200 lines of C with ctypes bindings. This is the kind of optimization that only matters at >100 pages/s throughput, which is exactly where the v2 architecture is heading after ONNX quantization.

```python
    """Find connections within N degrees of separation"""
    query = """
    WITH RECURSIVE company_graph AS (
        -- Base case: start with target company
        SELECT 
            c.id,
            c.name,
            0 as depth,
            CAST(c.id AS TEXT) as path
        FROM companies c
        WHERE c.id = ?
        
        UNION ALL
        
        -- Recursive case: follow edges
        SELECT 
            CASE 
                WHEN e.source_id = cg.id THEN e.target_id
                ELSE e.source_id
            END as id,
            c.name,
            cg.depth + 1 as depth,
            cg.path || '->' || CAST(
                CASE 
                    WHEN e.source_id = cg.id THEN e.target_id
                    ELSE e.source_id
                END AS TEXT

        FROM company_graph cg
        JOIN edges e ON e.source_id = cg.id OR e.target_id = cg.id
        JOIN companies c ON c.id = CASE 
            WHEN e.source_id = cg.id THEN e.target_id
            ELSE e.source_id

            AND cg.id != CASE

                ELSE e.source_id
```

```python
@dataclass
class PipelineTask:
    name: str
    coroutine: callable
    dependencies: List[str]
    timeout: float = 30.0
    retry_count: int = 3

class StructuredPipelineOrchestrator:
    def __init__(self, max_concurrent: int = 10):
        self.max_concurrent = max_concurrent
        self.task_registry: Dict[str, PipelineTask] = {}
        self.task_graph: Dict[str, List[str]] = {}
        
    def register_task(self, task: PipelineTask):
        """Register a pipeline task with dependencies"""
        self.task_registry[task.name] = task
        self.task_graph[task.name] = task.dependencies
        
    async def execute_pipeline(self, pipeline_name: str, input_data: dict):
        """Execute pipeline with structured concurrency"""
        async with anyio.create_task_group() as tg:
            # Create execution context
            context = {
                "pipeline_start": time.time(),
                "input_data": input_data,
                "results": {},
                "errors": {}
            }
            
            # Execute tasks with dependency resolution
            execution_order = self._topological_sort()
            
            for task_name in execution_order:
                task = self.task_registry[task_name]
                
                # Check dependencies are complete
                for dep in task.dependencies:
                    if dep not in context["results"]:
                        await self._wait_for_dependency(dep, context)
                
                # Execute task with structured error handling
                tg.start_soon(
                    self._execute_task_with_retry,
                    task,
                    context
                )
            
            # Wait for completion with timeout
            try:
                await asyncio.wait_for(tg.__aenter__(), timeout=300.0)
            except asyncio.TimeoutError:
                self._handle_timeout(context)
                
        return context["results"]
    
    async def _execute_task_with_retry(self, task: PipelineTask, context: dict):
        """Execute task with retry logic and cancellation support"""
        for attempt in range(task.retry_count):
            try:
                # Create cancellation scope
                async with anyio.move_on_after(task.timeout) as scope:
                    result = await task.coroutine(
                        context["input_data"],
                        **{dep: context["results"][dep] for dep in task.dependencies}
                    )
                    
                    if scope.cancel_called:
                        raise TimeoutError(f"Task {task.name} timed out")
                    
                    context["results"][task.name] = result
                    return
                    
            except Exception as e:
                if attempt == task.retry_count - 1:
                    context["errors"][task.name] = str(e)
                    raise
                await asyncio.sleep(2 ** attempt)  # Exponential backoff
    
    def _topological_sort(self) -> List[str]:
        """Kahn's algorithm for topological sorting"""
        in_degree = {node: 0 for node in self.task_graph}
        for node in self.task_graph:
            for neighbor in self.task_graph[node]:
                in_degree[neighbor] = in_degree.get(neighbor, 0) + 1
        
        queue = [node for node in in_degree if in_degree[node] == 0]
        sorted_nodes = []
        
        while queue:
            node = queue.pop(0)
            sorted_nodes.append(node)
            
            for neighbor in self.task_graph.get(node, []):
                in_degree[neighbor] -= 1
                if in_degree[neighbor] == 0:
                    queue.append(neighbor)
        
        return sorted_nodes
```
